Converting integer binary and fraction binary back to decimal

;Integer Portion
mov bl, 10
mov si, 0
mov al, int_bineary_porttion
while(al > 0) {
  div bl
  bcd_integer_array[si] = ah
  inc si
}

;Fraction Portion
mov bl, 10
mov si, 0
mov al, int_bineary_porttion
while(al > 0 && si < max_digits) {
  mul bl
  bcd_fraction_array[si] = ah
  inc si
}
