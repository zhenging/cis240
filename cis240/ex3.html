Part 1 AX = (AL/BL) * CL

DIV BL
MUL CL

Part 2 AX = (AL/BL) * CL

# Step 1
AL/BL = AL + AH/BL
# Step 2:
(AL/BL) * CL = AL * CL + (AH/BL) * CL
             = AX * CL + (AH*CL) / BL

DIV BL
MOV DL, AL
MOV AL, AH
MUL CL
DIV BL
MOV DH, AL
MOV AL, DL
MUL CL
ADD AL, DH

Part 3 AX = (AX/BX) * CX

# Step 1
AX/BX = AX + DX/BX
# Step 2:
(AX/BX) * CX = AX * CX + (DX/BX) * CX
             = AX * CX + (DX * CX) / BX

DIV BX
MOV SI, AX
MOV AX, DX
MUL CX
DIV BX
MOV DI, AX
MOV AX, SI
MUL CX
ADD AX, DI
