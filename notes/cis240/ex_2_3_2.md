```
	DOSSEG
	.MODEL SMALL
	.STACK 100
	.DATA
	.CODE
MESSAGE PROC
	mov     ax,@DATA
	mov     ds,ax
        mov     bx,0200h
        mov     cx,0019h
        mov     si,0000
        mov     ah,02
LOOPTOP:
        mov     dl,[bx+si]
        int     21h
        inc     si
        dec     cx
        JNZ     LOOPTOP
        mov     ax,4C00h           ; Load DOS Exit function (4Ch)
                                       ;   in AH and 0 errorlevel in AL
        int     21h                ; Call DOS
MESSAGE ENDP
	END
```
